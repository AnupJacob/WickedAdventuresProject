Create Table Sripts
-------------------

CREATE TABLE T_WARIDEDETAILS(RIDE_NUMBER INT IDENTITY(10001,1) PRIMARY KEY, RIDE_NAME VARCHAR(50) NOT NULL,HEALTH_SECURITY INT NOT NULL,RIDE_COST FLOAT NOT NULL, RIDE_STATUS INT NOT NULL, CHANGED_DATE DATE NOT NULL DEFAULT GETDATE())

CREATE TABLE T_WACUSTOMERDETAILS(CUSTOMER_ID INT IDENTITY(20001,1) PRIMARY KEY, CUSTOMER_NAME VARCHAR(50) NOT NULL,CUSTOMER_ADDRESS VARCHAR(150) NOT NULL,CONTACT INT NOT NULL, EMAIL_ID VARCHAR(50) NOT NULL, PASSWD VARCHAR(50) NOT NULL, CHANGED_DATE DATE NOT NULL DEFAULT GETDATE())

CREATE TABLE T_WAPAYMENTDETAILS(PAYMENT_ID INT IDENTITY(30001,1),CUSTOMER_ID INT REFERENCES T_WACUSTOMERDETAILS(CUSTOMER_ID), RIDE_NUMBER INT REFERENCES T_WARIDEDETAILS(RIDE_NUMBER), BOOKING_DATE DATE NOT NULL DEFAULT GETDATE(),PAYMENT_STATUS INT NOT NULL,PAYMENT_METHOD VARCHAR(20) NOT NULL, CHANGED_DATE DATE NOT NULL DEFAULT GETDATE())
