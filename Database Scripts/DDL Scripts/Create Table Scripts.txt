Create Table Sripts
-------------------

CREATE TABLE T_WARIDEDETAILS(RIDE_NUMBER INT PRIMARY KEY, RIDE_NAME VARCHAR(20) NOT NULL,HEALTH_SECURITY INT NOT NULL,RIDE_COST FLOAT NOT NULL, RIDE_STATUS INT NOT NULL, CHANGED_DATE DATE NOT NULL DEFAULT GETDATE())

CREATE TABLE T_WACUSTOMERDETAILS(CUSTOMER_NUMBER INT PRIMARY KEY, CUSTOMER_NAME VARCHAR(50) NOT NULL,CUSTOMER_ADDRESS VARCHAR(150) NOT NULL,CONTACT INT NOT NULL, CHANGED_DATE DATE NOT NULL DEFAULT GETDATE())

CREATE TABLE T_WAPAYMENTDETAILS(PAYMENT_ID INT,CUSTOMER_NUMBER INT REFERENCES T_WACUSTOMERDETAILS(CUSTOMER_NUMBER), RIDE_NUMBER INT REFERENCES T_WARIDEDETAILS(RIDE_NUMBER), PAYMENT_STATUS INT NOT NULL,PAYMENT_METHOD VARCHAR(20) NOT NULL, CHANGED_DATE DATE NOT NULL DEFAULT GETDATE())
