Create Table Sripts
-------------------

CREATE TABLE T_WARIDEDETAILS(RIDE_NUMBER INT PRIMARY KEY, RIDE_NAME VARCHAR(20),HEALTH_SECURITY INT,RIDE_COST FLOAT, RIDE_STATUS INT, CHANGED_DATE DATE)

CREATE TABLE T_WACUSTOMERDETAILS(CUSTOMER_NUMBER INT PRIMARY KEY, CUSTOMER_NAME VARCHAR(50),CUSTOMER_ADDRESS VARCHAR(150),CONTACT INT, CHANGED_DATE DATE)

CREATE TABLE T_WAPAYMENTDETAILS(PAYMENT_ID INT,CUSTOMER_NUMBER INT REFERENCES T_WACUSTOMERDETAILS(CUSTOMER_NUMBER), RIDE_NUMBER INT REFERENCES T_WARIDEDETAILS(RIDE_NUMBER), PAYMENT_STATUS INT,PAYMENT_METHOD VARCHAR(20), CHANGED_DATE DATE)
